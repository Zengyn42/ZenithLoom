
class BitBoard:
    def __init__(self, w: int, h: int):
        self.W = w
        self.H = h
        self._full = (1 << (w * h)) - 1
        self.col0 = sum(1 << (r * w) for r in range(h))
        self.colW_minus_1 = sum(1 << (r * w + w - 1) for r in range(h))

    def _idx(self, r, c):
        return r * self.W + c

    def make_mask(self, cells):
        mask = 0
        for r, c in cells:
            mask |= (1 << self._idx(r, c))
        return mask

    def popcount(self, n):
        return bin(n).count('1')

    def flood_fill(self, origin, blocked):
        start_bit = 1 << self._idx(origin[0], origin[1])
        if blocked & start_bit:
            return 0
        
        visited = start_bit
        frontier = start_bit
        
        while True:
            next_frontier = (
                ((frontier << 1) & ~self.col0) |
                ((frontier >> 1) & ~self.colW_minus_1) |
                (frontier << self.W) |
                (frontier >> self.W)
            )
            next_frontier &= ~blocked & self._full
            next_frontier &= ~visited
            
            if not next_frontier:
                break
            
            visited |= next_frontier
            frontier = next_frontier
            
        return self.popcount(visited)

class GameEngine:
    @staticmethod
    def tick(state: 'GameState', dir_a: 'Direction', dir_b: 'Direction') -> 'GameState':
        new_state = copy.deepcopy(state)
        s_a = new_state.my_snake
        s_b = new_state.opponent
        
        # Phase 1: New heads
        dy_a, dx_a = dir_a.value
        dy_b, dx_b = dir_b.value
        new_head_a = (s_a.head[0] + dy_a, s_a.head[1] + dx_a)
        new_head_b = (s_b.head[0] + dy_b, s_b.head[1] + dx_b)
        
        alive_a = s_a.alive
        alive_b = s_b.alive
        
        # Phase 2: Wall kill
        if alive_a:
            if not (0 <= new_head_a[0] < new_state.board_height and 0 <= new_head_a[1] < new_state.board_width):
                alive_a = False
        if alive_b:
            if not (0 <= new_head_b[0] < new_state.board_height and 0 <= new_head_b[1] < new_state.board_width):
                alive_b = False
                
        # Phase 3: Head-to-head
        if alive_a and alive_b and new_head_a == new_head_b:
            if s_a.length < s_b.length:
                alive_a = False
            elif s_b.length < s_a.length:
                alive_b = False
            else:
                alive_a = False
                alive_b = False
                
        # Phase 4: Body collision
        def get_truncated_body(snake):
            # just_ate is state from the previous frame
            just_ate = getattr(snake, 'just_ate', False)
            body_list = list(snake.body)
            if not just_ate and len(body_list) > 0:
                return body_list[:-1]
            return body_list

        if alive_a:
            # New head in opponent's body -> opponent dies, attacker survives
            if new_head_a in get_truncated_body(s_b):
                alive_b = False
            # New head in own body -> self dies
            if new_head_a in get_truncated_body(s_a):
                alive_a = False
                
        if alive_b:
            # New head in opponent's body -> opponent dies, attacker survives
            if new_head_b in get_truncated_body(s_a):
                alive_a = False
            # New head in own body -> self dies
            if new_head_b in get_truncated_body(s_b):
                alive_b = False

        # Phase 5: Growth and Movement
        def resolve_move(snake, new_head, is_my_snake):
            nonlocal alive_a, alive_b
            is_alive = alive_a if is_my_snake else alive_b
            if not is_alive:
                snake.alive = False
                return
            
            just_ate = False
            if new_head in new_state.foods:
                just_ate = True
                new_state.foods.remove(new_head)
                new_food = GameEngine._spawn_food(new_state)
                if new_food:
                    new_state.foods.add(new_food)
            else:
                if len(snake.body) > 0:
                    snake.body.pop()
            
            snake.body.appendleft(new_head)
            snake.just_ate = just_ate
            snake.alive = True

        resolve_move(s_a, new_head_a, True)
        resolve_move(s_b, new_head_b, False)
        
        # Final status check (since resolve_move might set snake.alive=True even if alive_a was False)
        # Wait, if alive_a was False, resolve_move sets snake.alive=False.
        # If alive_a was True, resolve_move sets snake.alive=True.
        # The only thing is that a snake might have been killed in Phase 4.
        
        # Let's be explicit:
        s_a.alive = alive_a if (s_a.alive if hasattr(s_a, 'alive') else True) else False
        # Actually the resolve_move already handles it.
        
        new_state.frame += 1
        return new_state

    @staticmethod
    def _spawn_food(state: 'GameState') -> Optional[Tuple[int, int]]:
        occupied = set()
        if state.my_snake: occupied.update(state.my_snake.body_set)
        if state.opponent: occupied.update(state.opponent.body_set)
        occupied.update(state.foods)
        occupied.update(state.walls)
        
        for _ in range(200):
            y = random.randint(0, state.board_height - 1)
            x = random.randint(0, state.board_width - 1)
            if (y, x) not in occupied:
                return (y, x)
        return None
